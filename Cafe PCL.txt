}

Estoque{
----------------------------------------------------------------------
	double maxCafe, maxLeite, maxChocolate, maxCanela, maxAcucar;
----------------------------------------------------------------------
	void fazPedido(Bebida b);
	/* ELABORAR OS METODOS */

abstract Bebida{
-----------------------------------------------------------------
	double preco;
	int acucar;

}

}

CafeComLeite extends Cafe{

}
